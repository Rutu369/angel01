#include <iostream>
using namespace std;
int main() {
    int choice;
    cout << "Enter a number between 1 and 3: ";
    cin >> choice;
    switch (choice) {
        case 1:
            cout << "Welcome." << endl;
            break;
        case 2:
            cout << "Hello." << endl;
            break;
        case 3:
            cout << "How are you." << endl;
            break;
        default:
            cout << "Invalid choice. Please enter a number between 1 and 3." << endl;
    }
    return 0;
}
