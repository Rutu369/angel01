#include <iostream>
using namespace std;

class Employee {
private:
    string empName;
    double basicSalary;
    double bonus;
    double totalSalary;

    // Private function to calculate total salary
    void calculateTotal() {
        totalSalary = basicSalary + bonus;
    }

public:
    // Function to take input from user
    void inputDetails() {
        cout << "Enter Employee Name: ";
        cin >> empName;

        cout << "Enter Basic Salary: ";
        cin >> basicSalary;

        cout << "Enter Bonus: ";
        cin >> bonus;

        calculateTotal(); // compute total salary
    }

    // Function to display employee details
    void displayDetails() {
        cout << "\n--- Employee Details ---\n";
        cout << "Name: " << empName << endl;
        cout << "Basic Salary: " << basicSalary << endl;
        cout << "Bonus: " << bonus << endl;
        cout << "Total Salary: " << totalSalary << endl;
    }
};

int main() {
    Employee e1, e2; // multiple employee objects

    cout << "Enter details for Employee 1:\n";
    e1.inputDetails();

    cout << "\nEnter details for Employee 2:\n";
    e2.inputDetails();

    // Display details
    e1.displayDetails();
    e2.displayDetails();

    return 0;
}
