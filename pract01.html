#include #include using namespace std; class Student { private: string name; string yearOfAdmission; float marks; string address; public: void readDetails() { cout << "Enter Student Name: "; getline(cin, name); cout << "Enter Year of Admission: "; getline(cin, yearOfAdmission); cout << "Enter Marks: "; cin >> marks; cin.ignore(); // to clear newline after marks cout << "Enter Address: "; getline(cin, address); } void displayDetails() { cout << "\nStudent Information:\n"; cout << "Name: " << name << endl; cout << "Year of Admission: " << yearOfAdmission << endl; cout << "Marks: " << marks << endl; cout << "Address: " << address << endl; } }; int main() { Student s; s.readDetails(); s.displayDetails(); return 0; }
