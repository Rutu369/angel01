Write a Python program to check the consistency of the system of linear equation import numpy as np A = np.array([ [2, -1, 3], [4, 2, 6], [1, 1, 1] ]) B = np.array([5, 12, 3]) n = A.shape[1] AB = np.hstack((A, B.reshape(-1, 1))) rank_A = np.linalg.matrix_rank(A) rank_AB = np.linalg.matrix_rank(AB) print("Rank(A) =", rank_A) print("Rank([A|B]) =", rank_AB) if rank_A == rank_AB: print("\nsystem is CONSISTENT") if rank_A == n: print("-> Unique solution") else: print("-> Infinite solutions") else: print("\nsystem is INCONSISTENT") Rank(A) = 3 Rank([A|B]) = 3 system is CONSISTENT -> Unique solution
