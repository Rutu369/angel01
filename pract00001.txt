# Question: Write a Python code for Gaussâ€“Seidel Method

x = 0.0
y = 0.0
z = 0.0

n = 10   # number of iterations

print("Iteration\t x\t\t y\t\t z")

for i in range(n):
    x = (13 - 3*y - 2*z) / 8
    y = (7 - x - z) / 5
    z = (9 - 2*x - y) / 6
    print(f"{i+1}\t\t {x:.4f}\t {y:.4f}\t {z:.4f}")

print("\nApproximate solution after", n, "iterations :")
print(f"x = {x:.4f}  y = {y:.4f}  z = {z:.4f}")
